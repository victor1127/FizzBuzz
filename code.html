GetValues()

The main purpuse of this method is getting the initial values from the user, then it calls other methods to validate entries, generate the results and display them to the user. 


                        function GetValues()
                        {
                            let startValue= document.getElementById("startvalue").value;
                            let endValue= document.getElementById("endtvalue").value;
                            let isValidStartValue= IsValidEntry(startValue);
                            let isValidEndValue= IsValidEntry(endValue);
                        
                            if(!isValidStartValue || !isValidEndValue)
                            {
                                alert("Enter interger values only");
                                return;
                            }
                        
                            let results = GetFizzBuzz(startValue, endValue);
                            DisplayResults(results);
                            
                        }

IsValidEntry()

Validates the user entry, start and end values are intergers


                        function IsValidEntry(value)
                        {
                            let sValue = parseInt(value);
                            return (Number.isInteger(sValue));    
                        }

GetFizzBuzz()

This method evaluates weather a given number is deviseble by 3 or 5, in case it is devisible by 3 it will store 'Fizz' in the array instead of the number, or 'Buzz' if devisible by 5, in case of being devisible by both, it will store 'FizzBuzz', otherwise, will store the actual value before returning an array. 


                        function GetFizzBuzz(value,endValue)
                        {
                            let output=[];
                            let message="";
                        
                            for(let i=value;i<=endValue;i++)
                            {
                                if(i%3==0) message+="Fizz";
                                if(i%5==0) message+="Buzz";
                                if(message=="") message=i;    
                                
                                output.push(message);
                                message="";
                            }
                        
                            return output;
                        
                        }

DisplayResults()

DisplayResults is a method that iterates throught a given array to assign values to a table which will display the results supplied by the 'GetFizzBuzz' method. To assign the values to each table data this function make use of another method named 'FillOutTableData()'. 


                        function DisplayResults(results)
                        {
                            let tbdody = document.getElementById("tableresult");
                            let template =  document.getElementById("fbtemplate");
                            tbdody.innerHTML="";
                        
                            for(var i = 0; i < results.length; i+=5)
                            {
                                let tableRow = document.importNode(template.content,true);
                                let rowCols =  tableRow.querySelectorAll("td");
                                FillOutTableData(rowCols,results,i);
                                tbdody.appendChild(tableRow);
                            }             
                        }

FillOutTableData()

This function takes 3 parameters,'Array: all the table data for the table', 'Array: results', 'int: index'. It will iterate throught the number of table data assigning a value and a CSS class to each one. 


                        function FillOutTableData(rowCols,results,index)
                        {
                            for(var i=0; i < rowCols.length; i++)
                            {
                                rowCols[i].classList.add(results[index]);
                                rowCols[i].textContent = results[index];
                                index++;
                            }
                        
                        }
